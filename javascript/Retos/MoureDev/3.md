/*
 * Escribe un programa que muestre cÃ³mo transcurre un juego de tenis y quiÃ©n lo ha ganado.
 * El programa recibirÃ¡ una secuencia formada por "P1" (Player 1) o "P2" (Player 2), segÃºn quien
 * gane cada punto del juego.
 *
 * - Las puntuaciones de un juego son "Love" (cero), 15, 30, 40, "Deuce" (empate), ventaja.
 * - Ante la secuencia [P1, P1, P2, P2, P1, P2, P1, P1], el programa mostrarÃ­a lo siguiente:
 *   15 - Love
 *   30 - Love
 *   30 - 15
 *   30 - 30
 *   40 - 30
 *   Deuce
 *   Ventaja P1
 *   Ha ganado el P1
 * - Si quieres, puedes controlar errores en la entrada de datos.
 * - Consulta las reglas del juego si tienes dudas sobre el sistema de puntos.

const puntuaciones = ["Love", "15", "30", "40"];

const overQuar = (p1, p2) => {
  if (p1 == p2) return "Deuce";
  if (p1 > p2 ? p1 - p2 >= 2 : p2 - p1 >= 2)
    return `Ha ganado el ${p1 > p2 ? "P1" : "P2"}`;
  return `Ventaja ${p1 > p2 ? "P1" : "P2"}`;
};

const juego = (jogo) => {
  let timeLine = [];
  let win = false;
  let points = {
    P1: 0,
    P2: 0,
  };

  jogo.forEach((e) => {
    if (win) return false;
    if (points.P1 == 5 | points.P2 == 5){
        timeLine.push["Deuce"]
        return false;
    }
    if (points.P1 > 5 | points.P2 > 5){
        timeLine = ["Bad data"]
        win = true
        return false;
    }
    points[e] += 1;

    if (points.P1 >= 4 || points.P2 >= 4) {
      timeLine.push(overQuar(points.P1, points.P2));
      win =
        points.P1 > points.P2
          ? points.P1 - points.P2 >= 2
          : points.P2 - points.P1 >= 2;
      return false;
    }

    timeLine.push(`${puntuaciones[points.P1]} - ${puntuaciones[points.P2]} `);
  });

  return timeLine.join("\n");
};
